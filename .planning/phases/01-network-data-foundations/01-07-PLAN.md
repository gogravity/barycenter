      ```python
      """Stable JSON canonicalization helper. Used by chain.canonicalize_json."""
      import json
      from datetime import datetime, date
      from uuid import UUID
      from typing import Any

      def _default(obj: Any) -> Any:
          if isinstance(obj, UUID):
              return str(obj)
          if isinstance(obj, (datetime, date)):
              return obj.isoformat()
          raise ValueError(f"Cannot canonicalize value of type {type(obj).__name__}")

      def canonicalize(obj: dict) -> str:
          return json.dumps(
              obj,
              sort_keys=True,
              separators=(",", ":"),
              ensure_ascii=False,
              default=_default,
          )
      ```

      ```python
      """AuditClient — fail-closed audit emitter (D-04, D-06)."""
      import logging
      from contextlib import contextmanager
      from barycenter.audit.models import AuditEvent
      from barycenter.audit.sinks import LogsAnalyticsSink, WormBlobSink
      from barycenter.audit.exceptions import AuditEmitError, FailClosedAbort
      from barycenter.audit.chain import (
          canonicalize_json, compute_digest, read_head_locked, update_head,
      )

      log = logging.getLogger(__name__)

      class AuditClient:
          def __init__(self, sql_conn, la_sink: LogsAnalyticsSink, worm_sink: WormBlobSink):
              self._sql = sql_conn
              self._la = la_sink
              self._worm = worm_sink

          def emit(self, event: AuditEvent) -> AuditEvent:
              try:
                  cur = self._sql.cursor()
                  prior = read_head_locked(cur)
                  event.prior_digest = prior
                  payload_no_self = event.model_dump(mode="json", exclude={"this_digest"})
                  canonical_no_self = canonicalize_json(payload_no_self)
                  event.this_digest = compute_digest(prior, canonical_no_self)
                  payload_full = event.model_dump(mode="json")
                  canonical_full = canonicalize_json(payload_full)

                  # LA upload — must succeed
                  self._la.upload(event)
                  # WORM append — must succeed
                  self._worm.append(canonical_full.encode("utf-8"))
                  # chain_state update — must succeed
                  update_head(cur, event.this_digest)
                  # Commit only after all three sinks succeed
                  self._sql.commit()
                  return event
              except Exception as exc:
                  try:
                      self._sql.rollback()
                  except Exception as rollback_exc:
                      log.error("rollback failed during fail-closed abort: %r", rollback_exc)
                  raise FailClosedAbort(f"audit emit failed: {exc!r}") from exc

          @contextmanager
          def recording_query(self, actor_id: str, query_description: str, tenant_id: str | None = None):
              """AUDIT-02: emit an audit-read event around a query against the audit chain.
              Yields control to the caller; emits success or failure on exit.
              """
              from datetime import datetime, timezone
              from uuid import uuid4
              outcome = "success"
              error_repr: str | None = None
              try:
                  yield
              except Exception as e:
                  outcome = "failure"
                  error_repr = repr(e)
                  raise
              finally:
                  try:
                      self.emit(AuditEvent(
                          event_id=uuid4(),
                          occurred_at=datetime.now(timezone.utc),
                          actor_id=actor_id,
                          actor_type="service",
                          verb="audit.read",
                          resource_type="audit.chain_state",
                          outcome=outcome,
                          tenant_id=tenant_id,
                          metadata={"query": query_description, "error": error_repr},
                      ))
                  except Exception as audit_err:
                      log.error("audit-of-audit emit failed: %r", audit_err)
                      # Re-raise only if the original block didn't already raise
                      if outcome == "success":
                          raise
      ```

      ```python
      from barycenter.audit.exceptions import ChainIntegrityError

      def validate_chain(canonical_entries: list[str]) -> int:
          """Recompute the chain from GENESIS_HASH; raise ChainIntegrityError on first mismatch.
          Each entry is the canonical JSON of an AuditEvent (already including this_digest field).
          Returns the count of validated entries on success.
          """
          import json
          prior = GENESIS_HASH
          count = 0
          for raw in canonical_entries:
              data = json.loads(raw)
              claimed_prior = data.get("prior_digest")
              claimed_this = data.get("this_digest")
              if claimed_prior != prior:
                  raise ChainIntegrityError(
                      f"entry {count} prior_digest mismatch: claimed {claimed_prior!r}, expected {prior!r}"
                  )
              # Recompute this_digest using canonicalize_json on the payload without this_digest
              payload_no_self = {k: v for k, v in data.items() if k != "this_digest"}
              canonical_no_self = canonicalize_json(payload_no_self)
              recomputed = compute_digest(prior, canonical_no_self)
              if recomputed != claimed_this:
                  raise ChainIntegrityError(
                      f"entry {count} this_digest mismatch: claimed {claimed_this!r}, recomputed {recomputed!r}"
                  )
              prior = claimed_this
              count += 1
          return count
      ```
