    """AUDIT-03: WORM container 6-year retention is locked at container creation
    and cannot be shortened by any role.
    """
    import os
    import subprocess
    import json
    import pytest

    AZURE_SUB = os.environ.get("AZURE_SUBSCRIPTION_ID")
    RG_AUDIT = os.environ.get("RG_AUDIT", "rg-barycenter-dev")
    STORAGE_ACCT = os.environ.get("WORM_STORAGE_ACCOUNT", "stbarywormdev")
    CONTAINER = "audit"
    EXPECTED_RETENTION_DAYS = 2190

    pytestmark = pytest.mark.skipif(not AZURE_SUB, reason="AZURE_SUBSCRIPTION_ID not set")

    def _az_json(*args) -> dict:
        result = subprocess.run(["az", *args, "--output", "json"],
                                capture_output=True, text=True, check=True)
        return json.loads(result.stdout) if result.stdout.strip() else {}

    def test_immutability_policy_locked_at_6_years():
        policy = _az_json(
            "storage", "container", "immutability-policy", "show",
            "--account-name", STORAGE_ACCT,
            "--container-name", CONTAINER,
        )
        assert policy.get("state") == "Locked", f"WORM policy state: {policy.get('state')}"
        assert policy.get("immutabilityPeriodSinceCreationInDays") == EXPECTED_RETENTION_DAYS, (
            f"Retention days: {policy.get('immutabilityPeriodSinceCreationInDays')} != {EXPECTED_RETENTION_DAYS}"
        )
        assert policy.get("allowProtectedAppendWrites") is True, "Append writes must be allowed during retention"

    def test_locked_policy_cannot_be_shortened():
        """Attempt to PATCH retention to 30 days. Must fail."""
        result = subprocess.run(
            ["az", "storage", "container", "immutability-policy", "extend",
             "--account-name", STORAGE_ACCT,
             "--container-name", CONTAINER,
             "--period", "30"],
            capture_output=True, text=True,
        )
        # Azure refuses to shorten a locked policy; must observe nonzero exit
        assert result.returncode != 0, (
            "Locked WORM policy was shortenable! AUDIT-03 broken. stdout: "
            f"{result.stdout!r} stderr: {result.stderr!r}"
        )

    def test_storage_account_is_private():
        acct = _az_json("storage", "account", "show",
                        "--name", STORAGE_ACCT, "--resource-group", RG_AUDIT)
        assert acct["publicNetworkAccess"] == "Disabled", acct.get("publicNetworkAccess")
        assert acct["allowBlobPublicAccess"] is False, acct.get("allowBlobPublicAccess")
        assert acct["allowSharedKeyAccess"] is False, "AAD-only access required"
        assert acct["minimumTlsVersion"] == "TLS1_2"
        assert acct["networkRuleSet"]["defaultAction"] == "Deny"

    def test_test_container_account_is_deleted():
        """Pitfall 7: after test-container lock validation, the test storage account
        must no longer exist before plan 01 phase exits."""
        result = subprocess.run(
            ["az", "storage", "account", "show",
             "--name", os.environ.get("WORM_TEST_STORAGE_NAME", "stbarywormtest1"),
             "--resource-group", RG_AUDIT,
             "--output", "json"],
            capture_output=True, text=True,
        )
        # Expect failure (account does not exist)
        assert result.returncode != 0, (
            "Test WORM storage account still exists after Pitfall 7 validation. "
            "Delete it manually before phase exit."
        )
