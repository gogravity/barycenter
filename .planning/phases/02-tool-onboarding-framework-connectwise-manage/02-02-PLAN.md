    ```python
    """PrimitiveResult dataclass — return type of every primitive."""
    from __future__ import annotations
    from dataclasses import dataclass, field
    from typing import Any

    VALID_FIELD_CLASSES = frozenset({"RESTRICTED", "SENSITIVE", "INTERNAL", "PUBLIC", "DROPPED"})

    @dataclass(frozen=True)
    class PrimitiveResult:
        expr: str
        params: dict[str, Any] = field(default_factory=dict)
        field_class: str = "INTERNAL"

        def __post_init__(self) -> None:
            if self.field_class not in VALID_FIELD_CLASSES:
                raise ValueError(
                    f"field_class must be one of {VALID_FIELD_CLASSES}, got {self.field_class!r}")
    ```

    ```python
    """aggregate primitive: numeric reduction over a list of values."""
    import statistics
    from barycenter.etl.primitives._result import PrimitiveResult

    _FNS = {
        "SUM": sum,
        "COUNT": len,
        "AVG": lambda xs: statistics.mean(xs) if xs else 0,
        "MAX": max,
        "MIN": min,
    }

    def aggregate(field: str, fn: str, values: list) -> PrimitiveResult:
        if fn not in _FNS:
            raise ValueError(f"aggregate fn must be one of {set(_FNS)}, got {fn!r}")
        result = _FNS[fn](values) if values else 0
        return PrimitiveResult(expr="?", params={field: result}, field_class="INTERNAL")
    ```

    `primitives/bucket.py`:
    ```python
    """bucket primitive: map a numeric value to a labeled bucket range."""
    from barycenter.etl.primitives._result import PrimitiveResult

    def bucket(field: str, value, ranges: list) -> PrimitiveResult:
        # ranges = [10, 50, 100] -> labels: "<10", "10-50", "50-100", ">100"
        if value is None:
            label = "null"
        else:
            label = None
            sorted_r = sorted(ranges)
            if value < sorted_r[0]:
                label = f"<{sorted_r[0]}"
            elif value >= sorted_r[-1]:
                label = f">={sorted_r[-1]}"
            else:
                for lo, hi in zip(sorted_r, sorted_r[1:]):
                    if lo <= value < hi:
                        label = f"{lo}-{hi}"
                        break
        return PrimitiveResult(
            expr="?",
            params={f"{field}_bucket": label},
            field_class="INTERNAL")
    ```

    `primitives/score.py`:
    ```python
    """score primitive: numeric formula evaluated over fields. RESTRICTED inputs forbidden."""
    from barycenter.etl.primitives._result import PrimitiveResult

    _ALLOWED = set("0123456789+-*/(). ")

    def score(fields: dict, formula: str) -> PrimitiveResult:
        # Sanitize formula: only operators + numbers + field names
        safe = formula
        for name in fields:
            safe = safe.replace(name, str(fields[name]))
        if not all(c in _ALLOWED for c in safe):
            raise ValueError(f"score formula contains disallowed chars: {formula!r} -> {safe!r}")
        result = eval(safe, {"__builtins__": {}}, {})  # restricted eval, numeric only
        return PrimitiveResult(expr="?", params={"score": result}, field_class="INTERNAL")
    ```

    Step 3 — Create `framework/recipe.py`:

    ```python
    """ETLRecipe: the declarative composition of primitives that produces a parameterized SQL.

    A recipe is a mapping from target column -> (primitive_name, kwargs). At compile time,
    every derivation MUST resolve to a member of barycenter.etl.primitives.PRIMITIVE_REGISTRY.
    The CI gate test_recipe_no_bypass.py walks every recipe and asserts this invariant.
    """
    from __future__ import annotations
    from typing import Any
    from pydantic import BaseModel, ConfigDict, field_validator

    class ETLRecipe(BaseModel):
        model_config = ConfigDict(extra="forbid", frozen=False)

        target_table: str
        # column_name -> (primitive_name, kwargs_dict)
        derivations: dict[str, tuple[str, dict[str, Any]]]

        @field_validator("derivations")
        @classmethod
        def validate_no_bypass(cls, v: dict) -> dict:
            from barycenter.etl.primitives import PRIMITIVE_REGISTRY
            for col, deriv in v.items():
                if not (isinstance(deriv, tuple) and len(deriv) == 2):
                    raise ValueError(
                        f"derivation for {col!r} must be (primitive_name, kwargs); got {deriv!r}")
                primitive_name, _kwargs = deriv
                if primitive_name not in PRIMITIVE_REGISTRY:
                    raise ValueError(
                        f"recipe column {col!r} bypasses primitive layer: "
                        f"{primitive_name!r} not in PRIMITIVE_REGISTRY")
            return v

        def compile(self, record: dict, *, kv_client=None, tenant_id: str | None = None
                   ) -> tuple[str, dict[str, Any]]:
            """Apply each primitive against record, build INSERT projection.

            Returns (sql_template, params_dict). The caller is responsible for binding
            params and executing; primitives never execute SQL themselves.
            """
            from barycenter.etl.primitives import PRIMITIVE_REGISTRY
            cols: list[str] = []
            exprs: list[str] = []
            params: dict[str, Any] = {}
            for col, (primitive_name, kwargs) in self.derivations.items():
                fn = PRIMITIVE_REGISTRY[primitive_name]
                # Resolve `field` source from the record where applicable
                source_field = kwargs.get("field")
                value = _resolve_field(record, source_field) if source_field else None
                # Dispatch to the primitive with kwargs adjusted for its signature
                if primitive_name == "drop":
                    res = fn(source_field or col)
                elif primitive_name == "hash":
                    res = fn(col, value if value is not None else "")
                elif primitive_name == "pseudonymize":
                    res = fn(col, value or "", tenant_id or "", kv_client,
                             kwargs.get("salt_version"))
                elif primitive_name == "as_is":
                    res = fn(col, value if value is not None else kwargs.get("default"),
                             only_classes=tuple(kwargs.get("only_classes", ("PUBLIC", "INTERNAL"))),
                             field_class=kwargs.get("field_class", "PUBLIC"))
                elif primitive_name == "bucket":
                    res = fn(col, value, kwargs.get("ranges", []))
                elif primitive_name == "aggregate":
                    res = fn(col, kwargs.get("fn", "SUM"), value if isinstance(value, list) else [value])
                elif primitive_name == "score":
                    res = fn(kwargs.get("fields", {col: value or 0}), kwargs.get("formula", "0"))
                elif primitive_name == "keyword_flags":
                    res = fn(col, value or "", kwargs.get("kw_dict", {}))
                else:
                    raise ValueError(f"unhandled primitive {primitive_name!r}")
                if res.field_class == "DROPPED":
                    continue
                cols.append(col)
                exprs.append(res.expr)
                for k, val in res.params.items():
                    params[k] = val
            sql = f"INSERT INTO {self.target_table} ({', '.join(cols)}) VALUES ({', '.join(exprs)})"
            return sql, params


    def _resolve_field(record: dict, path: str):
        """Resolve dotted field paths like 'company.id' or '_info.lastUpdated'."""
        if not path:
            return None
        cur = record
        for part in path.split("."):
            if "[]" in part:
                # Skip array projection in compile (used by keyword_flags as text dump)
                key = part.replace("[]", "")
                items = cur.get(key, []) if isinstance(cur, dict) else []
                if isinstance(items, list):
                    return " ".join(str(i.get("name", i)) if isinstance(i, dict) else str(i) for i in items)
                return None
            if isinstance(cur, dict):
                cur = cur.get(part)
            else:
                return None
            if cur is None:
                return None
        return cur


    def iter_all_recipes():
        """Discover all recipe factories under barycenter.etl.adapters.*. Used by no-bypass CI gate.

        Plan 05 populates the connectwise.recipes module; until then this is empty.
        """
        recipes = []
        try:
            from barycenter.etl.adapters.connectwise.recipes import (
                companies, agreements, tickets, configurations, time_entries
            )
            for mod in (companies, agreements, tickets, configurations, time_entries):
                for attr in dir(mod):
                    if attr.endswith("_recipe"):
                        recipes.append(getattr(mod, attr)())
        except ImportError:
            pass
        return recipes
    ```

    Step 4 — Create `framework/pseudonymizer.py`:

    ```python
    """Pseudonymizer: HMAC-SHA256(salt, email_lower) -> person_pid.

    Pitfall 5 mitigation: salt is fetched per call from Key Vault, used in the HMAC,
    and dereferenced via `del`. NEVER assigned to a module/class attribute. NEVER appears
    in __repr__, logging, or AuditEvent payloads.
    """
    from __future__ import annotations
    import hmac
    import hashlib

    class Pseudonymizer:
        def __init__(self, kv_client) -> None:
            self._kv = kv_client

        def derive(self, email: str, tenant_id: str,
                   salt_version: str | None = None) -> tuple[str, str]:
            if not email or not isinstance(email, str):
                raise ValueError(f"email must be a non-empty string, got {email!r}")
            if not tenant_id or not isinstance(tenant_id, str):
                raise ValueError(f"tenant_id must be a non-empty string, got {tenant_id!r}")
            secret_name = f"salt-{tenant_id}"
            secret = self._kv.get_secret(secret_name, version=salt_version) \
                if salt_version is not None else self._kv.get_secret(secret_name)
            try:
                salt_material = secret.value
                if isinstance(salt_material, str):
                    salt_bytes = salt_material.encode("utf-8")
                else:
                    salt_bytes = salt_material
                pid = hmac.new(
                    salt_bytes,
                    email.lower().encode("utf-8"),
                    hashlib.sha256
                ).hexdigest()
                version = secret.properties.version
                return pid, version
            finally:
                # Pitfall 5: dereference salt material immediately
                try:
                    del salt_bytes
                except UnboundLocalError:
                    pass
                del secret

        def __repr__(self) -> str:
            # Salt material must NEVER appear here.
            return f"<Pseudonymizer kv={type(self._kv).__name__}>"
    ```
